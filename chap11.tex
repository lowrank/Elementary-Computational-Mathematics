\begin{problem}[Peierls–Bogoliubov]
Prove the following inequality 
    \begin{equation*}
        \tr{e^A e^B} \ge \tr{e^{A + B}} \ge e^f
    \end{equation*}
    if $A, B$ are Hermitian matrices and $\tr{e^B} = 1$, $f = \tr{A e^B}$.
\end{problem}
